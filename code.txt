using System;
using System.Collections.Generic;

namespace PC
{
    class Component
    {
        public string Model { get; set; }

        private double power;
        public double PowerConsumption
        {
            get => power;
            set => power = value < 0 ? 0 : value;
        }

        public Component(string model, double power)
        {
            Model = model;
            PowerConsumption = power;
        }
    }

    class PCAssembly
    {
        public int MaxSlots { get; set; }
        private List<Component> components;

        public PCAssembly(int maxSlots)
        {
            MaxSlots = maxSlots;
            components = new List<Component>();
        }

        public void AddComponent(Component comp)
        {
            if (components.Count < MaxSlots)
                components.Add(comp);
            else
                Console.WriteLine($"Свободных слотов нет! Не удалось добавить: {comp.Model}");
        }

        public void ShowTotalPower()
        {
            double total = 0;
            foreach (var c in components)
            {
                Console.WriteLine($"Деталь: {c.Model}, Потребление: {c.PowerConsumption} Вт");
                total += c.PowerConsumption;
            }
            Console.WriteLine($"Общее энергопотребление сборки: {total} Вт");
        }
    }

    class Program
    {
        static void Main()
        {
            Console.Write("Введите количество слотов: ");
            int max = int.Parse(Console.ReadLine());

            PCAssembly pc = new PCAssembly(max);

            for (int i = 1; i <= 3; i++)
            {
                Console.Write($"Название детали {i}: ");
                string model = Console.ReadLine();

                Console.Write($"Энергопотребление (Вт): ");
                double power = double.Parse(Console.ReadLine());

                pc.AddComponent(new Component(model, power));
            }

            Console.WriteLine("\n--- Итоговый отчет ---");
            pc.ShowTotalPower();
        }
    }
}
